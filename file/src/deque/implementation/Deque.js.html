Deque.prototype.insert = function ( i , x ) {

	this._checkbounds( i ) ;

	this.append( x ) ;

	let j = this.len( ) - 1 ;

	for ( ; i < j ; --j ) {

		const a = this.get( j ) ;
		this.set( j , this.get( j - 1 ) ) ;
		this.set( j - 1 , a ) ;

	}

	return this ;

} ;

	return this ;

} ;

} ;

		this.set( i , b ) ;

} ;
